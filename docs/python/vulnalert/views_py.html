  86    def login_form(request, template='login.html', next=''):
  87        """
            Processes original login to create a user on the site
            once validated successfully logs the user in and redirects to default user page
            """

  91        response_dict = {}
  92        (response_dict['message'], response_dict['error_message']) = ('','')
  93        response_dict['debug'] = 'yes'
  94        next = request.REQUEST.get('next', next)
  95        if request.method == "POST":
  96            response_dict['form'] = LoginForm(request.POST)
  97            if not request.session.test_cookie_worked():
  98                response_dict['error_message'] = _("Looks like your browser isn't configured to accept cookies. Please enable cookies, reload this page, and try again.")
  99                request.session.set_test_cookie()
 100                return render_to_response(template, response_dict)
 101            else:
 102                is_valid = response_dict['form'].is_valid()
 103                if is_valid:
 104                    data = request.POST.copy()
 105                    user = authenticate(username=data['username'], password=data['password'])
 106                    if user:
 107                        login( request, user )
 108                        user_profile = request.user.get_profile()
 109                        criteria_pref_type = user_profile.criteria_pref_type
 110                        next = '../prefs_' + criteria_pref_type + '/'
 111                        return HttpResponseRedirect( next )
 112                if not is_valid or not user:
 113                    """ Only return a generic error msg here for security -- otherwise they'd know if username or password was correct """
 114                    response_dict['error_message'] = _("Please enter a correct username and password. Note that both fields are case-sensitive.")
 115                
 116        else:
 117            response_dict['form'] = LoginForm()
 118        if request.session.get('from_signup', None):
 119            del request.session['from_signup']
 120            response_dict['message'] = "Signup successful! Please login to continue"
 121        request.session.set_test_cookie()
 122        return render_to_response(template, response_dict, context_instance=RequestContext(request))

 178    def user_prefs_simple(request, template='prefs_simple.html', adv_url='../prefs_advanced'):
 179        """ 
            Allow user to modify their preferences, specifically:
                Sources to search
                Alert Delivery Methods
                Criteria to search for (each set of values that make up a criterion is one instance of the criteria form)
            """

 185        
 186        # initialize vars
 187        response_dict = { 'forms':{} }
 188        response_dict['error_message'] = ''
 189        criteria_list = []
 190        criteria_valid_dict = {}
 191        criteria_valid_list = []
 192    
 193        if request.method == "POST":
 194            valid = True
 195    
 196            for input_name in request.POST.keys():
 197                # deal with multiple copies of the criteria form -- prefix allows each set to have a distinct naming scheme
 198                if input_name.startswith("criteria"):
 199                    prefix = input_name.split('-')[0]
 200                    input_value = request.POST.get(input_name, None)
 201                    criteria = {'value':input_value}
 202                    
 203                    # django bug #3969: very unfortunately django's newforms cannot handle a situation where a post returns both any errors and the values submitted when using name prefixes
 204                    # so we'll have to create a list of forms for display (criteria_list), and a dict for validation
 205                    # addendum -- forms class totally fails if bound to a dict and prefix is set -- apparently loses track of the name<->form_var mapping so going to have to explicitly create
 206                    # errors...
 207                    # update: once newforms.formsets is finalized (i.e. available in trunk instead of just newforms-admin branch) -- this will change to use that much cleaner scheme
 208                    criteria_form_valid = CriteriaFormSimple( criteria )
 209    
 210                    # transform things so the errors show up
 211                    criteria_form = CriteriaFormSimple( initial=criteria, prefix=prefix )
 212                    if criteria_form_valid.errors != {}:
 213                        valid = False
 214                        for key in criteria_form_valid.errors:
 215                            criteria_form.errors[key] = criteria_form_valid.errors[key]
 216                    else:
 217                        # append a tuple of (prefix, form) since we can't get the prefix otherwise
 218                        criteria_valid_list.append( (prefix, criteria_form_valid) )
 219    
 220                    criteria_list.append( criteria_form )
 221    
 222            # update the response (will propogate to render at end of function)
 223            response_dict['forms']['criteria_list'] = criteria_list
 224            response_dict['forms']['prefs'] = PreferencesForm(request.POST)
 225    
 226            if valid and response_dict['forms']['prefs'].is_valid():
 227                # set simple as default if enable exists in post, else disable it
 228                profile = request.user.get_profile()
 229                if request.POST.has_key('enable'):
 230                    profile.criteria_pref_type = 'simple'
 231                else:
 232                    profile.criteria_pref_type = 'advanced'
 233                profile.save()
 234                response_dict['forms']['prefs'].save(request.user)
 235                for (prefix, criteria) in criteria_valid_list:
 236                    type   = prefix.split('_')[1]
 237                    criteria_type = CriteriaType.objects.get(name=type) # name/version
 238                    qualifier     = CriteriaQualifier.objects.get(name='contains')
 239                    if prefix.split('_')[-1] == 'new' and criteria.clean_data['value'] != '':
 240                        # create new criteria
 241                        user_criteria = UserCriteria( user=request.user, criteria_type=criteria_type, qualifier=qualifier, value=criteria.clean_data['value'] )
 242                        user_criteria.save()
 243                    elif prefix.split('_')[-1] == 'new' and criteria.clean_data['value'] == '':
 244                        continue
 245                    elif criteria.clean_data['value'] == '':
 246                        # user set existing criteria to blank so delete it from db
 247                        criteria_id = prefix.split('_')[-1]
 248                        user_criteria = UserCriteria.objects.get( id=criteria_id )
 249                        user_criteria.delete()
 250                    else:
 251                        # modify existing criteria
 252                        criteria_id = prefix.split('_')[-1]
 253                        user_criteria = UserCriteria.objects.get( id=criteria_id)
 254                        user_criteria.value = criteria.data['value'] # this makes no sense... everything validates but clean_data is not populated, no biggie, just weird
 255                        user_criteria.save()
 256                request.user.message_set.create(message="Preferences Update Successfully!")
 257                # reload the newly entered/changed database values
 258                return HttpResponseRedirect( "." )
 259                
 260        else:
 261            # initial load so grab from user's prefs in db
 262            user_prefs_dict = {}
 263            user_profile = request.user.get_profile()
 264            user_prefs_dict['alert_methods'] = [k['name'] for k in user_profile.alert_methods.values()]
 265            user_prefs_dict['sources'] = [k['id'] for k in user_profile.sources.values()]
 266            if user_profile.criteria_pref_type == 'simple':
 267                user_prefs_dict['enable'] = True
 268    
 269            user_criteria_objs = UserCriteria.objects.filter( user=request.user )
 270            user_criteria_dict = {}
 271            for user_criteria in user_criteria_objs:
 272                criteria_type_name = user_criteria.criteria_type.name
 273                if criteria_type_name in SIMPLE_CRITERIA_TYPES:
 274                    criteria_list.append(CriteriaFormSimple( initial={'value':user_criteria.value, }, prefix="criteria_"+criteria_type_name+'_'+str(user_criteria.id)) )
 275    
 276            # update the response dict
 277            response_dict['forms']['criteria_list'] = criteria_list
 278            response_dict['forms']['prefs'] = PreferencesForm( initial=user_prefs_dict )
 279    
 280        #response_dict['forms']['criteria_list'] = [ CriteriaFormSimple( {'value':'bleh'})  ]
 281        response_dict['advanced_prefs_url'] = adv_url
 282        return render_to_response(template, response_dict, context_instance=RequestContext(request))

 316                    criteria_dict[ criteria_id ][ input_type ] = request.POST[ input_name ]
